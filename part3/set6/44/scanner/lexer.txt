%class-name Scanner
%filenames scanner

linenr      "#>"[0-9]+\n
ws          [ \t]+
id          [a-zA-Z_][a-zA-Z_0-9]*
int         -?[0-9]+
double      -?[0-9]+\.[0-9]+

%%

{ws}        return WS;
^{linenr}   {
                d_line = stoul(matched().substr(2));
                return LINENR;
            }
{id}        return IDENT;
{int}       return INT;
{double}    return DOUBLE;
\n          return NEWLINE;
.           return CHAR;

%%
